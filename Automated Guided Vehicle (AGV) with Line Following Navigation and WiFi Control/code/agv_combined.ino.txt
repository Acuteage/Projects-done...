#include <ESP8266WiFi.h>
#include <ESP8266WebServer.h>

/* ------------ WIFI SETTINGS ------------ */
const char* ssid = "AGV_Robot";
const char* password = "12345678";

ESP8266WebServer server(80);

/* ------------ MOTOR PINS (L298N) ------------ */
#define IN1 D1
#define IN2 D2
#define IN3 D3
#define IN4 D4
#define ENA D5
#define ENB D6

/* ------------ IR SENSOR PINS ------------ */
#define LEFT_SENSOR  D7
#define RIGHT_SENSOR D8

/* ------------ MODE VARIABLE ------------ */
String mode = "AUTO";   // AUTO or MANUAL
String command = "STOP";

/* ------------ MOTOR FUNCTIONS ------------ */
void forward() {
  digitalWrite(IN1,HIGH);
  digitalWrite(IN2,LOW);
  digitalWrite(IN3,HIGH);
  digitalWrite(IN4,LOW);
}

void backward() {
  digitalWrite(IN1,LOW);
  digitalWrite(IN2,HIGH);
  digitalWrite(IN3,LOW);
  digitalWrite(IN4,HIGH);
}

void left() {
  digitalWrite(IN1,LOW);
  digitalWrite(IN2,HIGH);
  digitalWrite(IN3,HIGH);
  digitalWrite(IN4,LOW);
}

void right() {
  digitalWrite(IN1,HIGH);
  digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW);
  digitalWrite(IN4,HIGH);
}

void stopMotor() {
  digitalWrite(IN1,LOW);
  digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW);
  digitalWrite(IN4,LOW);
}

/* ------------ LINE FOLLOWING ------------ */
void lineFollower() {

  int leftSensor  = digitalRead(LEFT_SENSOR);
  int rightSensor = digitalRead(RIGHT_SENSOR);

  if(leftSensor == LOW && rightSensor == LOW)
    forward();

  else if(leftSensor == HIGH && rightSensor == LOW)
    left();

  else if(leftSensor == LOW && rightSensor == HIGH)
    right();

  else
    stopMotor();
}

/* ------------ WEB CONTROL ------------ */
void handleRoot() {

  String page =
  "<h1>AGV Robot Control</h1>"
  "<a href='/auto'>AUTO MODE</a><br><br>"
  "<a href='/manual'>MANUAL MODE</a><br><br>"
  "<a href='/F'>Forward</a><br>"
  "<a href='/B'>Backward</a><br>"
  "<a href='/L'>Left</a><br>"
  "<a href='/R'>Right</a><br>"
  "<a href='/S'>Stop</a>";

  server.send(200,"text/html",page);
}

void setAuto(){
  mode="AUTO";
  server.send(200,"text/plain","AUTO MODE");
}

void setManual(){
  mode="MANUAL";
  server.send(200,"text/plain","MANUAL MODE");
}

void cmdF(){command="F";}
void cmdB(){command="B";}
void cmdL(){command="L";}
void cmdR(){command="R";}
void cmdS(){command="S";}

/* ------------ SETUP ------------ */
void setup() {

  Serial.begin(115200);

  pinMode(IN1,OUTPUT);
  pinMode(IN2,OUTPUT);
  pinMode(IN3,OUTPUT);
  pinMode(IN4,OUTPUT);
  pinMode(ENA,OUTPUT);
  pinMode(ENB,OUTPUT);

  pinMode(LEFT_SENSOR,INPUT);
  pinMode(RIGHT_SENSOR,INPUT);

  analogWrite(ENA,1023);
  analogWrite(ENB,1023);

  /* WIFI START */
  WiFi.softAP(ssid,password);
  Serial.println("WiFi Started");

  server.on("/",handleRoot);
  server.on("/auto",setAuto);
  server.on("/manual",setManual);

  server.on("/F",[]{cmdF(); server.send(200,"text/plain","Forward");});
  server.on("/B",[]{cmdB(); server.send(200,"text/plain","Backward");});
  server.on("/L",[]{cmdL(); server.send(200,"text/plain","Left");});
  server.on("/R",[]{cmdR(); server.send(200,"text/plain","Right");});
  server.on("/S",[]{cmdS(); server.send(200,"text/plain","Stop");});

  server.begin();
}

/* ------------ LOOP ------------ */
void loop() {

  server.handleClient();

  if(mode == "AUTO") {
    lineFollower();
  }
  else if(mode == "MANUAL") {

    if(command=="F") forward();
    else if(command=="B") backward();
    else if(command=="L") left();
    else if(command=="R") right();
    else stopMotor();
  }
}
